(a) x1 = -1, y1 = 5, m = -2
    y - y1 = m(x - x1)
    y - 5  = -2(x + 1)
    y - 5  = -2x - 2
    y      = -2x - 7
    2x + y + 7 = 0

(c) y = 7 - 3x ; m = -3
    y1 = 5, x1 = 0; m = -3

    2y - 2  = -x + 1

## Excercise 3

(a) p(x) = x - 3

(b) p(x) = x - 3/2

(c) x^2 + 4x - 5   = (x - 1).p(x)
    (x + 5)(x - 1) = (x - 1).p(x)
    p(x) = x + 5

(d) x^3 + 4x^2 - 8 = (x + 2).p(x)
    x^3 + 4x^2 + (4x - 4x) - 8 = (x + 2).p(x)
    (x^2 + 2x)(x + 2) - 4x - 8 = (x + 2).p(x)
    x^2 + 2x + (-4x - 8/x + 2) = p(x)
    p(x) = x^2 + 2x - 4

(e) x^8 - 1 = (x - 1).p(x)
    x^8 - (1)^8 = (x - 1).p(x)
    (x - 1)^8 = (x - 1).p(x)
    p(x) = (x - 1)^7

(f) x^4 - 2x^2 + 1 = (x^2 - 2x + 1).p(x)
    (x^2 + 2x + 1)(x^2 - 2x + 1) = (x^2 - 2x + 1).p(x)
    p(x) = x^2 + 2x + 1
